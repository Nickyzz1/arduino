#include <Arduino.h>

#include <VirtualWire.h>

void setup()
{
	vw_set_tx_pin(5);
	Serial.begin(9600);
	Serial.println("setup");
	vw_set_ptt_inverted(true);
	vw_setup(2000);
	pinMode(13, OUTPUT);
}

void loop()
{
	byte x[1];
	x[0] = 0;
	x[0] = analogRead(A0) >> 2;
	digitalWrite(13, HIGH);
	vw_send(x, 1);
	vw_wait_tx();
	digitalWrite(13, LOW);
}
